logs/misbehaviors.log → Logstash → Elasticsearch → report.py → Kibana dashboard

python detector.py data/<bsm_file>
# Restart Logstash to re-index
docker compose restart logstash
# Re-import the updated dashboard
curl -s -X POST "http://localhost:5601/api/saved_objects/_import?overwrite=true" -H "kbn-xsrf: true" -F file=@elk/kibana/kpi-vega.ndjson | python3 -m json.tool

##############  ELK STACK  ##############
docker compose down
docker compose up -d
# Force recreate
docker-compose up --force-recreate setup

##############  FILEBEAT  ##############
# Recreate Filebeat (new config + clean registry)
docker compose rm -sf filebeat && docker compose up -d filebeat
# Restart Filebeat
docker compose restart filebeat

##############  LOGSTASH ####### #######
# Stop Logstash
docker compose stop logstash
# Start Logstash
docker compose start logstash
# Restart Logstash
docker compose restart logstash
# Confirm Logstash is ready
curl -s http://localhost:9600/ | python3 -m json.tool | grep status
# Watch Logstash come up — takes ~60s
docker compose logs -f logstash | grep -E "Pipeline|pipeline|misbehavior|error|Error"
# Clear the Logstash data
docker volume rm mbd_logstash_data
docker compose run --rm logstash rm /usr/share/logstash/data/sincedb_misbehaviors
docker exec mbd-logstash rm -f /usr/share/logstash/data/sincedb_misbehaviors

##############  ELASTIC SEARCH  ##############
# 1. Delete the data
curl -X DELETE http://localhost:9200/mbd-misbehaviors-2026.02.27
# Update the ES index template
curl -X PUT http://localhost:9200/_index_template/mbd-misbehaviors  -H "Content-Type: application/json"   -d @elk/elasticsearch/index-template.json
# Delete the existing indices so they are recreated with the correct mapping
curl -X DELETE "http://localhost:9200/mbd-misbehaviors-*"
# Query data
curl -s http://localhost:9200/mbd-misbehaviors-*/_count
# List the index name
curl http://localhost:9200/_cat/indices?v
# Then delete by exact name, e.g.:
curl -X DELETE http://localhost:9200/mbd-misbehaviors-2026.02.27
